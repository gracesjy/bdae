## 디버깅을 위한 클래스

자동 소멸자를 이용해서 Redirection 부분 (print 만으로도 로깅 위한) 을 닫아 주도록 했다.<br>

```
import logging
import contextlib
from StreamToLogger import StreamToLogger
import sys
import logging, logging.handlers

class BDAELogger:
    def __init__(self, name="bdaeLogger", level="DEBUG"):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(logging.DEBUG)
        self.name = name
        self.level = level

        self.socketHandler = logging.handlers.SocketHandler('localhost',
                logging.handlers.DEFAULT_TCP_LOGGING_PORT)
        self.logger.addHandler(self.socketHandler)
        sys.stdout = StreamToLogger(self.logger,logging.INFO)
        sys.stderr = StreamToLogger(self.logger,logging.ERROR)

    def warning(self, msg):
        self.logger.warning(msg)

    def info(self, msg):
        self.logger.info(msg)

    def error(self, msg):
        self.logger.error(msg)

    def debug(self, msg):
        self.logger.debug(msg)

    def flush(self): pass
    def close(self):
        self.socketHandler.close()
        self.logger.removeHandler(self.socketHandler)

    def __del__(self):
        self.socketHandler.close()
        self.logger.removeHandler(self.socketHandler)
```
